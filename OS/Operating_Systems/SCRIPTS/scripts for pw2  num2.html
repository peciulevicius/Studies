    while read line
	do
		echo "Adding $line"
		arguments=( $line )
	name=${arguments[0]}
	lastname=${arguments[1]}
	surname=${arguments[2]}
	fullname=""
	if [ surname = "" ] 
	then
		surname=$lastname
		fullname="$name.$surname"
	else
		fullname="$name-$lastname.$surname"
	fi

	home="/home/students/$ou/$fullname"

	if [ ! -d $home ]
	then
		mkdir $home
		chmod 700 -R $home

		useradd -b "/home/students/$ou" -d $home -G $OU,students -p "linuxPower" $name
	else
		echo "User $name already exists"
	fi
	done < users.txt
